"""
nanoLlama - a complete Llama-style LLM in pure PyTorch.
Trains on CPU in ~20 minutes. Generates Shakespeare-like text.
215,000 parameters. Written from scratch across 15 lessons.
"""
import torch, math, time
import torch.nn as nn
import torch.nn.functional as F
from dataclasses  import dataclass
from torch.utils.data import Dataset, DataLoader


# ── Config ────────────────────────────────────────────────────
@dataclass
class Config:
    vocab_size:  int   = 256
    max_seq_len: int   = 128
    d_model:     int   = 64
    n_heads:     int   = 4
    n_layers:    int   = 4
    d_ff:        int   = 172
    rope_base:   float = 10000.0
    def __post_init__(self): self.d_head = self.d_model // self.n_heads


# ── Dataset ───────────────────────────────────────────────────
class TextDataset(Dataset):
    def __init__(self, ids, seq_len):
        self.ids, self.seq_len = ids, seq_len
    def __len__(self): return len(self.ids) - self.seq_len
    def __getitem__(self, i):
        x = self.ids[i   : i+self.seq_len]
        y = self.ids[i+1 : i+self.seq_len+1]
        return torch.tensor(x, dtype=torch.long), torch.tensor(y, dtype=torch.long)


# ── Model Components (Lessons 06–09) ─────────────────────────
class RMSNorm(nn.Module):
    def __init__(self, d, eps=1e-6):
        super().__init__(); self.eps = eps; self.w = nn.Parameter(torch.ones(d))
    def forward(self, x):
        return x / x.pow(2).mean(-1, keepdim=True).add(self.eps).sqrt() * self.w

def rope_freqs(d_head, max_seq, base=10000.0):
    i = torch.arange(0, d_head, 2).float()
    f = torch.outer(torch.arange(max_seq).float(), 1.0 / (base ** (i/d_head)))
    return f.cos(), f.sin()

def apply_rope(x, cos, sin):
    e, o = x[..., 0::2], x[..., 1::2]
    c = cos[:x.shape[-2]].unsqueeze(0).unsqueeze(0)
    s = sin[:x.shape[-2]].unsqueeze(0).unsqueeze(0)
    return torch.stack([e*c-o*s, e*s+o*c], dim=-1).flatten(-2)

class MHA(nn.Module):
    def __init__(self, cfg):
        super().__init__()
        self.H, self.Dh = cfg.n_heads, cfg.d_head
        D = cfg.d_model
        self.Wq = nn.Linear(D,D,bias=False); self.Wk = nn.Linear(D,D,bias=False)
        self.Wv = nn.Linear(D,D,bias=False); self.Wo = nn.Linear(D,D,bias=False)
        c, s = rope_freqs(cfg.d_head, cfg.max_seq_len, cfg.rope_base)
        self.register_buffer('cos', c); self.register_buffer('sin', s)
    def forward(self, x):
        B,S,D = x.shape; H,Dh = self.H, self.Dh
        q = apply_rope(self.Wq(x).view(B,S,H,Dh).transpose(1,2),self.cos,self.sin)
        k = apply_rope(self.Wk(x).view(B,S,H,Dh).transpose(1,2),self.cos,self.sin)
        v =            self.Wv(x).view(B,S,H,Dh).transpose(1,2)
        sc = (q@k.transpose(-2,-1))/math.sqrt(Dh)
        sc = sc.masked_fill(torch.triu(torch.ones(S,S,device=x.device,dtype=torch.bool),1),float('-inf'))
        o  = F.softmax(sc,dim=-1)@v
        return self.Wo(o.transpose(1,2).contiguous().view(B,S,D))

class FFN(nn.Module):
    def __init__(self, cfg):
        super().__init__()
        self.W1=nn.Linear(cfg.d_model,cfg.d_ff,bias=False)
        self.W3=nn.Linear(cfg.d_model,cfg.d_ff,bias=False)
        self.W2=nn.Linear(cfg.d_ff,cfg.d_model,bias=False)
    def forward(self, x): return self.W2(F.silu(self.W1(x))*self.W3(x))

class Block(nn.Module):
    def __init__(self, cfg):
        super().__init__()
        self.attn=MHA(cfg); self.ffn=FFN(cfg)
        self.n1=RMSNorm(cfg.d_model); self.n2=RMSNorm(cfg.d_model)
    def forward(self, x):
        x = x + self.attn(self.n1(x))
        x = x + self.ffn(self.n2(x))
        return x

class NanoLlama(nn.Module):
    def __init__(self, cfg):
        super().__init__(); self.cfg = cfg
        self.embed  = nn.Embedding(cfg.vocab_size, cfg.d_model)
        self.blocks = nn.ModuleList([Block(cfg) for _ in range(cfg.n_layers)])
        self.norm   = RMSNorm(cfg.d_model)
        self.head   = nn.Linear(cfg.d_model, cfg.vocab_size, bias=False)
        self.head.weight = self.embed.weight  # weight tying
        self._init()
    def _init(self):
        nn.init.normal_(self.embed.weight, 0, 0.02)
        for b in self.blocks:
            for l in [b.attn.Wq,b.attn.Wk,b.attn.Wv,b.attn.Wo,b.ffn.W1,b.ffn.W2,b.ffn.W3]:
                nn.init.normal_(l.weight, 0, 0.02)
            nn.init.normal_(b.attn.Wo.weight, 0, 0.02/self.cfg.n_layers**0.5)
            nn.init.normal_(b.ffn.W2.weight,  0, 0.02/self.cfg.n_layers**0.5)
    def forward(self, ids):
        x = self.embed(ids)
        for b in self.blocks: x = b(x)
        return self.head(self.norm(x))
    def n_params(self): return sum(p.numel() for p in self.parameters())


# ── LR Schedule ───────────────────────────────────────────────
def get_lr(step, total, peak=3e-4, mn=3e-5, warmup=100):
    if step < warmup: return peak * (step+1) / warmup
    p = (step-warmup)/(total-warmup)
    return mn + (peak-mn) * 0.5 * (1 + math.cos(math.pi * p))


# ── Generate ──────────────────────────────────────────────────
def generate(model, ids, n=200, T=0.8, topk=40):
    model.eval()
    g = ids.clone()
    with torch.no_grad():
        for _ in range(n):
            logits = model(g[:, -model.cfg.max_seq_len:])[:, -1, :] / T
            if topk:
                v, _ = torch.topk(logits, topk)
                logits[logits < v[:, [-1]]] = float('-inf')
            g = torch.cat([g, torch.multinomial(F.softmax(logits,-1),1)], dim=1)
    return g


# ── Main Training ─────────────────────────────────────────────
if __name__ == '__main__':
    torch.manual_seed(42)
    cfg   = Config()
    model = NanoLlama(cfg)
    print(f"nanoLlama: {model.n_params():,} parameters")

    # Download TinyShakespeare if needed
    import urllib.request, os
    if not os.path.exists('shakespeare.txt'):
        url = 'https://raw.githubusercontent.com/karpathy/char-rnn/master/data/tinyshakespeare/input.txt'
        urllib.request.urlretrieve(url, 'shakespeare.txt')
        print("Downloaded TinyShakespeare")

    text   = open('shakespeare.txt').read()
    ids    = list(text.encode('utf-8'))
    n      = len(ids)
    tr_ids = ids[:int(n*0.9)]
    vl_ids = ids[int(n*0.9):]
    tr_dl  = DataLoader(TextDataset(tr_ids, cfg.max_seq_len), batch_size=16, shuffle=True,  drop_last=True)
    vl_dl  = DataLoader(TextDataset(vl_ids, cfg.max_seq_len), batch_size=16, shuffle=False, drop_last=True)

    opt    = torch.optim.AdamW(model.parameters(), lr=3e-4, weight_decay=0.1)
    n_epochs = 10
    total_steps = len(tr_dl) * n_epochs
    global_step = 0
    best_val = float('inf')

    for epoch in range(n_epochs):
        model.train()
        tr_loss, nb = 0.0, 0
        for x, y in tr_dl:
            lr = get_lr(global_step, total_steps)
            for g in opt.param_groups: g['lr'] = lr
            B,S,_ = (logits := model(x)).shape
            loss = F.cross_entropy(logits.view(B*S,-1), y.view(B*S))
            opt.zero_grad(); loss.backward()
            torch.nn.utils.clip_grad_norm_(model.parameters(), 1.0)
            opt.step()
            tr_loss += loss.item(); nb += 1; global_step += 1

        model.eval()
        vl_loss, nb2 = 0.0, 0
        with torch.no_grad():
            for x, y in vl_dl:
                B,S,_ = (logits := model(x)).shape
                vl_loss += F.cross_entropy(logits.view(B*S,-1), y.view(B*S)).item(); nb2 += 1
        tr = tr_loss/nb; vl = vl_loss/nb2
        print(f"Epoch {epoch+1:2d} | train={tr:.3f} val={vl:.3f} ppl={math.exp(vl):.1f} lr={lr:.2e}")
        if vl < best_val:
            best_val = vl; torch.save({'model': model.state_dict(), 'cfg': cfg}, 'nanollama.pt')
            print("  ✓ saved")

        # Generate a sample after each epoch
        prompt = torch.tensor([[72,65,77]], dtype=torch.long)  # "Ham"
        out    = generate(model, prompt, n=150, T=0.8, topk=40)
        print("  Sample:", bytes(out[0].tolist()).decode('utf-8', errors='replace')[:120])
